Algoritmo distancia_entre_puntos
    Definir distancia, x1, x2, y1, y2, z1, z2 Como Numerica;;
    Escribir "Ingrese las coordenadas del primer punto:";
    Leer x1, y1, z1;
    Escribir "Ingrese las coordenadas del segundo punto:";
    Leer x2, y2 , z1;
    distancia ← RAIZ((x1-x2)↑2 + (y1-y2)↑2 + (z1-z2)↑2) ;
    Escribir "La distancia entre los dos puntos es:", distancia;
FinAlgoritmo
