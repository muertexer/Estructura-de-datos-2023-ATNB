Proceso numeros_primos
    Definir i, j, contador Como Entero;
    Definir es_primo Como Logico;
    Definir primos Como Cadena;
    
    primos ← "";
    Para i ← 2 Hasta 500 Con Paso 1 Hacer
        es_primo ← Verdadero;
        j ← 2;
        Mientras (j < i) Y es_primo Hacer
            Si i MOD j = 0 Entonces
                es_primo ← Falso;
            FinSi
            j ← j + 1;
        FinMientras
        
        Si es_primo Entonces
            primos ← primos + " " + ConvertirATexto(i);
            contador ← contador + 1;
        FinSi
        
        Si contador MOD 5 = 0 Entonces
            primos ← primos + "\n";
        FinSi
        
    FinPara
    
    Escribir("La cantidad de números primos entre 1 y 500 es:");
    Escribir contador;
    Escribir(primos);
FinProceso
